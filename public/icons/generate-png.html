PWA 아이콘 PNG 생성

이 페이지를 열고 개발자 도구 콘솔에서 다음 코드를 실행하세요:

// SVG를 Canvas로 변환하여 PNG 생성
const svg = document.querySelector('svg');
const canvas = document.createElement('canvas');
const ctx = canvas.getContext('2d');
const img = new Image();

img.onload = function() {
    // 192x192 생성
    canvas.width = 192;
    canvas.height = 192;
    ctx.drawImage(img, 0, 0, 192, 192);
    const link192 = document.createElement('a');
    link192.download = 'icon-192.png';
    link192.href = canvas.toDataURL('image/png');
    link192.click();
    
    // 512x512 생성
    canvas.width = 512;
    canvas.height = 512;
    ctx.drawImage(img, 0, 0, 512, 512);
    const link512 = document.createElement('a');
    link512.download = 'icon-512.png';
    link512.href = canvas.toDataURL('image/png');
    link512.click();
};

img.src = 'data:image/svg+xml;base64,' + btoa(document.querySelector('svg').outerHTML);
